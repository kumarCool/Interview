namespace MvcApplication.Models
{
    public class Employee
    {
        public int Age { get; set; }
        public String Name { get; set; }
        public String Address { get; set; }
    }
}


    public class EmployeeBinder : IModelBinder
    {
        public object BindModel(ControllerContext controllerContext,ModelBindingContext bindingContext)
        {
            var request = controllerContext.HttpContext.Request;
            string first_name = request.Form.Get("first_name");
            string middle_name = request.Form.Get("middle_name");
            string last_name = request.Form.Get("last_name");
            int Age = Convert.ToInt32(request.Form.Get("age"));
            string Address = request.Form.Get("EmployeeAddress");
            return new Employee
                { Name = first_name + " " + middle_name + " "+ last_name, 
                Age = Age, 
                Address = Address };
        }
    }

  public class EmployeeController : Controller
    {
        //
        // GET: /Employee/
        public ActionResult Index()
        {
            return View();
        }
        public ActionResult GetEmployeinfo([ModelBinder(typeof(EmployeeBinder))]Employee obj)
        {
            return View("Employee");
        }

    }
